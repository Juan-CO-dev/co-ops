```ts
import type { ParMode } from "@/lib/types";

/** A day-picked override row, as the resolver needs it. */
export interface ItemOverride {
  parValue: number | null;
  parUnit: string | null;
  parMode: ParMode;
}

/** Operational day-of-week (0–6, Sunday=0) from a YYYY-MM-DD operational date. */
export function operationalDayOfWeek(operationalDate: string): number {
  // Parse as a date-only value; append midday to avoid TZ edge rollovers.
  return new Date(`${operationalDate}T12:00:00`).getDay();
}

/**
 * Pick the override row for a day: specific-day row, else all-days base, else null.
 * `rows` = all active item_par_levels rows for one (item, location).
 */
export function pickOverride(
  rows: Array<{ dayOfWeek: number | null; parValue: number | null; parUnit: string | null; parMode: ParMode }>,
  dayOfWeek: number,
): ItemOverride | null {
  const specific = rows.find((r) => r.dayOfWeek === dayOfWeek);
  const base = rows.find((r) => r.dayOfWeek === null);
  const row = specific ?? base ?? null;
  if (!row) return null;
  return { parValue: row.parValue, parUnit: row.parUnit, parMode: row.parMode };
}

/**
 * Batch-load active override rows for items at one location, grouped by item_id.
 * (Item par overrides are location-scoped; the caller passes the resolved location.)
 */
export async function loadItemOverrides(
  service: SupabaseClient,
  itemIds: string[],
  locationId: string,
): Promise<Map<string, ItemOverride[]>> {
  const map = new Map<string, ItemOverride[]>();
  const ids = [...new Set(itemIds.filter((x): x is string => !!x))];
  if (ids.length === 0) return map;
  const { data, error } = await service
    .from("item_par_levels")
    .select("item_id, day_of_week, par_value, par_unit, par_mode")
    .in("item_id", ids)
    .eq("location_id", locationId)
    .eq("active", true);
  if (error) throw new Error(`loadItemOverrides: ${error.message}`);
  for (const r of (data ?? []) as Array<{ item_id: string; day_of_week: number | null; par_value: number | null; par_unit: string | null; par_mode: ParMode }>) {
    const list = map.get(r.item_id) ?? [];
    list.push({ dayOfWeek: r.day_of_week, parValue: r.par_value, parUnit: r.par_unit, parMode: r.par_mode } as ItemOverride & { dayOfWeek: number | null });
    map.set(r.item_id, list);
  }
  return map;
}
```

> NOTE: `loadItemOverrides`'s grouped rows need `dayOfWeek` for `pickOverride`. Define the map value as `Array<{ dayOfWeek: number|null; parValue; parUnit; parMode }>` (a small local type) rather than bare `ItemOverride[]` so `pickOverride` receives the day field. The implementer should pick one consistent row shape; the cleanest is: `loadItemOverrides` returns `Map<string, ItemParRow[]>` where `ItemParRow = { dayOfWeek; parValue; parUnit; parMode }`, and `pickOverride(rows, dow): ItemOverride | null` strips `dayOfWeek`. Adjust the snippet above to that single shape.
